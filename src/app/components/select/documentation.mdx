# Usage

  This component is very similar to radio list or checkbox list, but you need to choose options from dropdown. Instead of
  using child components you need to pass options using mandatory `options` prop.

  <Playground code={`import { Form, Select } from 'altrone-ui';

const OPTIONS = [
  { label: 'Afghanistan', value: 'Afghanistan' },
  { label: 'Albania', value: 'Albania' },
  { label: 'Algeria', value: 'Algeria' },
  ...
]

const [country, setCountry] = useState<string>();
const [visitedCountries, setVisitedCountries] = useState<string[]>([]);

return <Form>
  <Form.Field label="Your Motherland">
    <Select value={country} placeholder="Where have you born?" onChange={setCountry} options={OPTIONS} />
  </Form.Field>
  <Form.Field label="Visited Countries">
    <Select value={visitedCountries} multiple placeholder="What countries have you visited?" onChange={setVisitedCountries} options={OPTIONS} />
  </Form.Field>
</Form>
`}>
      <SelectDemo />
    </Playground>

</Text.Section>
